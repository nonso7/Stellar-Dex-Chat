# TypeScript SDK Examples

This guide demonstrates how to call the new contract functions (`heartbeat`, `deny_address`, `migrate_escrow`, `execute_batch_admin`) from the TypeScript SDK generated by Soroban contract bindings.

## Prerequisites

```typescript
import {
  Contract,
  TransactionBuilder,
  BASE_FEE,
  Networks,
  Address,
  nativeToScVal,
  scValToNative,
  rpc,
} from '@stellar/stellar-sdk';
```

## Helper Functions

```typescript
async function buildAndSimulate(
  publicKey: string,
  operation: ReturnType<Contract['call']>,
): Promise<{ assembledXdr: string }> {
  const account = await server.getAccount(publicKey);
  const tx = new TransactionBuilder(account, {
    fee: BASE_FEE,
    networkPassphrase: NETWORK_PASSPHRASE,
  })
    .addOperation(operation)
    .setTimeout(30)
    .build();

  const sim = await server.simulateTransaction(tx);
  if (rpc.Api.isSimulationError(sim)) {
    throw new Error(`Simulation failed: ${sim.error}`);
  }

  return {
    assembledXdr: rpc.assembleTransaction(tx, sim).build().toXDR(),
  };
}

async function submitAndWait(
  signedXdr: string,
  maxRetries: number = 20,
): Promise<string> {
  const tx = TransactionBuilder.fromXDR(signedXdr, NETWORK_PASSPHRASE);
  const sendResult = await server.sendTransaction(tx);
  if (sendResult.status === 'ERROR') {
    throw new Error(
      `Submission failed: ${JSON.stringify(sendResult.errorResult)}`,
    );
  }
  
  // Poll for confirmation
  let getResult = await server.getTransaction(sendResult.hash);
  let retries = 0;
  while (getResult.status === rpc.Api.GetTransactionStatus.NOT_FOUND) {
    if (retries >= maxRetries) {
      throw new Error(
        `Transaction confirmation timed out after ${maxRetries} attempts`,
      );
    }
    await new Promise((r) => setTimeout(r, 1500));
    retries += 1;
    getResult = await server.getTransaction(sendResult.hash);
  }
  
  if (getResult.status === rpc.Api.GetTransactionStatus.FAILED) {
    throw new Error('Transaction failed on-chain');
  }
  
  return sendResult.hash;
}
```

## 1. Heartbeat Function

The `heartbeat` function is called by operators to maintain their active status and includes replay protection via nonce validation.

### Function Signature
```rust
pub fn heartbeat(env: Env, operator: Address, nonce: u64) -> Result<(), Error>
```

### Replay Protection Architecture

For each operator, the contract keeps an independent nonce counter:

1. Client reads `get_operator_nonce(operator)` to get the **next expected nonce**
2. Client submits `heartbeat(operator, nonce)`
3. Contract accepts only exact matches, then increments the stored nonce
4. Reusing an old nonce fails with `StaleNonce`; skipping ahead fails with `InvalidNonce`

This design guarantees each signed operator action can be accepted only once.

### TypeScript Implementation

```typescript
interface HeartbeatParams {
  operatorAddress: string;
  nonce: number;
  adminPublicKey: string;
  signTx: (xdr: string) => Promise<string>;
}

export async function getOperatorNonce(operatorAddress: string): Promise<number> {
  const contract = new Contract(CONTRACT_ID);
  const op = contract.call('get_operator_nonce', new Address(operatorAddress).toScVal());
  const { assembledXdr } = await buildAndSimulate(DUMMY_SOURCE, op);
  const tx = TransactionBuilder.fromXDR(assembledXdr, NETWORK_PASSPHRASE);
  const sim = await server.simulateTransaction(tx);
  if (rpc.Api.isSimulationError(sim)) {
    throw new Error(`Failed to fetch operator nonce: ${sim.error}`);
  }
  const retval = (sim as rpc.Api.SimulateTransactionSuccessResponse).result?.retval;
  if (!retval) throw new Error('No nonce value returned');
  return Number(scValToNative(retval));
}

export async function submitHeartbeat(params: HeartbeatParams): Promise<string> {
  try {
    const { operatorAddress, nonce, adminPublicKey, signTx } = params;
    
    // Build the heartbeat operation
    const op = contract.call(
      'heartbeat',
      new Address(operatorAddress).toScVal(),
      nativeToScVal(nonce, { type: 'u64' }),
    );

    // Simulate and get assembled XDR
    const { assembledXdr } = await buildAndSimulate(adminPublicKey, op);
    
    // Sign and submit
    const signed = await signTx(assembledXdr);
    const hash = await submitAndWait(signed);
    
    console.log(`Heartbeat submitted successfully: ${hash}`);
    return hash;
    
  } catch (error) {
    console.error('Heartbeat failed:', error);
    
    // Handle specific error cases
    if (error.message.includes('NotOperator')) {
      throw new Error('Operator is not authorized. Please ensure the operator is properly set.');
    }
    if (error.message.includes('InvalidNonce')) {
      throw new Error('Invalid nonce. Please use a sequential nonce value.');
    }
    if (error.message.includes('StaleNonce')) {
      throw new Error('Stale nonce. This nonce has already been used.');
    }
    
    throw error;
  }
}

// Usage example

    nonce: nextNonce,
    adminPublicKey: 'GB...admin-public-key',
    signTx: async (xdr) => {
      // Implement your signing logic here
      return signedXdr;
    },
  });
  console.log('Heartbeat successful:', hash);
} catch (error) {

}
```

## 2. Deny Address Function

The `deny_address` function allows admins to add addresses to a denylist, preventing them from using the contract.

### Function Signature
```rust
pub fn deny_address(env: Env, address: Address) -> Result<(), Error>
```

### TypeScript Implementation

```typescript

  addressToDeny: string;
  adminPublicKey: string;
  signTx: (xdr: string) => Promise<string>;
}

  try {
    const { addressToDeny, adminPublicKey, signTx } = params;
    
    // Validate address format
    if (!addressToDeny.startsWith('G')) {
      throw new Error('Invalid Stellar address format');
    }

      'deny_address',
      new Address(addressToDeny).toScVal(),
    );

    // Simulate and get assembled XDR
    const { assembledXdr } = await buildAndSimulate(adminPublicKey, op);
    
    // Sign and submit
    const signed = await signTx(assembledXdr);
    const hash = await submitAndWait(signed);
    
    console.log(`Address denied successfully: ${hash}`);
    return hash;
    
  } catch (error) {
    console.error('Deny address failed:', error);
    
    // Handle specific error cases
    if (error.message.includes('Unauthorized')) {
      throw new Error('Only admin can deny addresses.');
    }
    if (error.message.includes('NotInitialized')) {
      throw new Error('Contract is not initialized.');
    }
    
    throw error;
  }
}
```

## 4. Execute Batch Admin Function

The `execute_batch_admin` function allows admins to execute multiple administrative operations in a single transaction.

### Function Signature
```rust
pub fn execute_batch_admin(env: Env, operations: Vec<BatchAdminOp>) -> Result<BatchResult, Error>
```

### TypeScript Implementation

```typescript

  op_type: string;
  payload: Buffer;

interface BatchResult {
  total_ops: number;
  success_count: number;
  failure_count: number;
  failed_index: number | null;
}

interface ExecuteBatchAdminParams {
  operations: BatchAdminOp[];
  adminPublicKey: string;
  signTx: (xdr: string) => Promise<string>;

  try {

    if (operations.length === 0) {

      throw new Error('Maximum 100 operations allowed per batch');
    }

      'execute_batch_admin',
      nativeToScVal(scOps, { type: 'Vec' }),
```

## Error Handling Patterns

### Common Error Types

```typescript
enum ContractError {
  Unauthorized = 201,
  NotInitialized = 101,
  InvalidRecipient = 204,
  NotOperator = 205,
  AddressDenied = 309,
  MigrationAlreadyComplete = 802,
  BatchOperationFailed = 803,
  InvalidNonce = 901,
  StaleNonce = 902,
}

function handleContractError(error: any): never {
  const message = error.message || '';
  
  if (message.includes('ContractError: 201')) {
    throw new Error('Unauthorized: Only authorized users can perform this action.');
  }
  if (message.includes('ContractError: 101')) {
    throw new Error('Contract not initialized. Please initialize the contract first.');
  }
  if (message.includes('ContractError: 205')) {
    throw new Error('Not an operator. Only operators can perform this action.');
  }
  if (message.includes('ContractError: 901')) {
    throw new Error('Invalid nonce. Please use a sequential nonce value.');
  }
  if (message.includes('ContractError: 902')) {
    throw new Error('Stale nonce. This nonce has already been used.');
  }
  
  // Generic error handling
  throw new Error(`Contract operation failed: ${message}`);
}
```

### Retry Pattern

```typescript
async function executeWithRetry<T>(
  operation: () => Promise<T>,
  maxRetries: number = 3,
  delayMs: number = 1000
): Promise<T> {
  let lastError: any;
  
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await operation();
    } catch (error) {
      lastError = error;
      
      // Don't retry on authorization errors
      if (error.message.includes('Unauthorized') || 
          error.message.includes('NotOperator')) {
        throw error;
      }
      
      if (attempt === maxRetries) {
        throw error;
      }
      
      // Exponential backoff
      const delay = delayMs * Math.pow(2, attempt - 1);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
  
  throw lastError;
}
```

## Best Practices

1. **Always validate inputs** before sending transactions
2. **Use proper error handling** to provide meaningful feedback to users
3. **Implement retry logic** for network-related failures
4. **Cache view-only results** to improve performance
5. **Always read nonce from chain right before signing** operator actions
6. **Monitor transaction status** and provide feedback to users
7. **Handle edge cases** like network timeouts and insufficient funds
8. **Validate admin permissions** before executing sensitive operations
